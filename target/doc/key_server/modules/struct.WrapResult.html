Converts from &mut Result<T, E> to Result<&mut T, &mut E>.

        Ok(v) => *v = 42,
        Err(e) => *e = 0,
    }
}

mutate(&mut x);
assert_eq!(x.unwrap(), 42);

mutate(&mut x);

pub fn iter_mut(&mut self) -> IterMut<'_, T>

Returns a mutable iterator over the possibly contained value.
